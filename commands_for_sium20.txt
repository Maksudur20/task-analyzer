# Commands for Sium20 to get the working Task Analyzer

# Step 1: Clone the working repository
git clone https://github.com/Maksudur20/task-analyzer.git

# Step 2: Navigate to the project
cd task-analyzer

# Step 3: Point to your repository
git remote set-url origin https://github.com/Sium20/Task_Analyzer.git

# Step 4: Push to your repository
git push -u origin main

# Step 5: Set up the project
dotnet restore
dotnet ef database update
dotnet run

# The application will be available at: http://localhost:5053
# Admin login: admin@taskanalyzer.com / Admin123!
